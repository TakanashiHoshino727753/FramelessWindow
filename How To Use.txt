一、把这两个文件（FramelessWindow.h FramelessWindow.cpp）放到你的项目根目录（和main.cpp同级）
二、修改CMakeLists.txt
1.在project(YourProjectName)后加上include_directories(${CMAKE_CURRENT_SOURCE_DIR})
2.把FramelessWindow加入编译
找到add_executable语句，在里面加上FramelessWindow.h和FramelessWindow.cpp 就像下面那样
add_executable(${PROJECT_NAME}
main.cpp
FramelessWindow.h#加这个
FramelessWindow.cpp#加这个
)
三、创建你的窗口子类（例如Form）
1.Form.h
#include"FramelessWindow.h"//包含基类头文件
class Form:public FramelessWindow//继承基类
...
public://构造函数：接收配置文件路径
explicit Form(const QString &configPath,QWidget *parent = nullptr);
...
2.Form.cpp
//在初始化列表里把配置路径传给基类
Form::Form(const QString &configPath,QWidget *parent):FramelessWindow(configPath,parent),ui(new Ui::Form)
...
四、main.cpp用法
int main(int argc,char *argv[ ]){
QApplication a(argc,argv);//设置配置文件路径（任意路径都可以）
QString configPath="window_config.ini";//创建窗口并传入路径
Form w(configPath);
w.show();
return a.exec();
}
